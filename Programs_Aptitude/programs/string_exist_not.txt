#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    string str[100],s;
    int i,len=10,count=0,j=0;
    while(1)
    {
    cout<<"Enter string and check whether it is exist or not"<<endl;
    for(i=0;i<len;i++)
    {
        cin>>s;
      if(str[i]==s)
      {
          cout<<"Given string is exist in an array"<<endl;
          break;
      }
      count++;
    }
    if(count==i)
    {
         str[j]=s;
        cout<<"String is insert in an array "<<str[j]<<endl;;
        j++;
    }
      string cmd;
      cin>>cmd;
      if(cmd=="exit")
      {
          break;
      }
      else if(cmd=="continue")
           {
               continue;
           }
    }
    
    return 0;
}
